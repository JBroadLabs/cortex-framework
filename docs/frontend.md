## 6. JavaScript Architecture

### 6.1 Module Overview

#### pieces.js - Piece Definitions
```javascript
// Piece class and factory
class Piece {
  constructor(type, color, position) {
    this.type = type; // 'pawn', 'rook', 'knight', 'bishop', 'queen', 'king'
    this.color = color; // 'white', 'black'
    this.position = position; // {file: 0-7, rank: 0-7}
    this.hasMoved = false;
  }

  getUnicodeSymbol() {
    const symbols = {
      white: { king: '♔', queen: '♕', rook: '♖', bishop: '♗', knight: '♘', pawn: '♙' },
      black: { king: '♚', queen: '♛', rook: '♜', bishop: '♝', knight: '♞', pawn: '♟' }
    };
    return symbols[this.color][this.type];
  }

  clone() {
    const cloned = new Piece(this.type, this.color, {...this.position});
    cloned.hasMoved = this.hasMoved;
    return cloned;
  }
}

// Export for use in other modules
if (typeof module !== 'undefined') {
  module.exports = Piece;
}
```

### 6.2 Event Handling

#### Click Event Flow
```javascript
handleSquareClick(file, rank) {
  const piece = this.engine.getPieceAt(file, rank);

  // If no piece selected
  if (!this.selectedSquare) {
    if (piece && piece.color === this.engine.currentTurn) {
      this.selectPiece(file, rank);
    }
    return;
  }

  // If piece already selected
  const from = this.selectedSquare;
  const to = {file, rank};

  // Try to move
  if (this.engine.isValidMove(from, to)) {
    this.engine.makeMove(from, to);
    this.deselectPiece();
    this.renderPieces();
    this.updateUI();
  } else if (piece && piece.color === this.engine.currentTurn) {
    // Select different piece
    this.deselectPiece();
    this.selectPiece(file, rank);
  } else {
    // Invalid move, deselect
    this.deselectPiece();
  }
}

selectPiece(file, rank) {
  this.selectedSquare = {file, rank};
  const validMoves = this.engine.getValidMoves(file, rank);
  this.highlightSelectedSquare(file, rank);
  this.highlightValidMoves(validMoves);
}

deselectPiece() {
  this.selectedSquare = null;

}
```

#### Button Event
```javascript
attachEventListeners() {
  // Board clicks (event delegation)
  this.boardElement.addEventListener('click', (e) => {
    const square = e.target.closest('.square');
    if (square) {
      const file = parseInt(square.dataset.file);
      const rank = parseInt(square.dataset.rank);
      this.handleSquareClick(file, rank);
    }
  });

  // New game button
  document.getElementById('new-game-btn').addEventListener('click', () => {
    this.resetGame();
  });
}
```

### 6.3 DOM Manipulation

#### Board Rendering
```javascript
renderBoard() {
  this.boardElement.innerHTML = '';

  // Render from rank 7 to 0 (top to bottom visually)
  for (let rank = 7; rank >= 0; rank--) {
    for (let file = 0; file < 8; file++) {
      const square = document.createElement('div');
      square.className = `square ${(rank + file) % 2 === 0 ? 'dark' : 'light'}`;
      square.dataset.file = file;
      square.dataset.rank = rank;

      const piece = this.engine.getPieceAt(file, rank);
      if (piece) {
        const pieceSpan = document.createElement('span');
        pieceSpan.className = `piece ${piece.color} ${piece.type}`;
        pieceSpan.textContent = piece.getUnicodeSymbol();
        square.appendChild(pieceSpan);
      }

      this.boardElement.appendChild(square);
    }
  }
}
```

#### UI Updates
```javascript
updateUI() {
  // Update turn display
  document.getElementById('turn-color').textContent =
    this.engine.currentTurn.charAt(0).toUpperCase() + this.engine.currentTurn.slice(1);

  // Update status
  const statusText = document.getElementById('status-text');
  statusText.className = '';

  if (this.engine.status === 'checkmate') {
    statusText.textContent = 'Checkmate!';
    statusText.className = 'checkmate';
  } else if (this.engine.status === 'stalemate') {
    statusText.textContent = 'Stalemate!';
  } else if (this.engine.status === 'check') {
    statusText.textContent = 'Check!';
    statusText.className = 'check';
    this.highlightKingInCheck();
  } else {
    statusText.textContent = 'In Progress';
  }
}
```

#### Highlighting
```javascript
highlightValidMoves(moves) {
  moves.forEach(({file, rank}) => {
    const square = this.getSquareElement(file, rank);
    square.classList.add('valid-move');

    // Add different class if square has enemy piece
    const piece = this.engine.getPieceAt(file, rank);
    if (piece) {
      square.classList.add('has-piece');
    }
  });
}

clearHighlights() {
  document.querySelectorAll('.square').forEach(square => {
    square.classList.remove('selected', 'valid-move', 'has-piece', 'in-check');
  });
}

highlightKingInCheck() {
  const king = this.engine.findKing(this.engine.currentTurn);
  if (king) {
    const square = this.getSquareElement(king.position.file, king.position.rank);
    square.classList.add('in-check');
  }
}
```

## 7. User Experience Flow

### 7.1 Game Initialization
1. Page loads
2. DOMContentLoaded fires
3. GameController.init() called
4. Board rendered with starting position
5. Event listeners attached
6. UI displays "White's turn"

### 7.2 Making a Move
1. User clicks white pawn at e2
2. Square highlights yellow
3. Valid moves (e3, e4) show green dots
4. User clicks e4
5. Pawn moves to e4
6. Highlights clear
7. Turn switches to Black
8. UI updates: "Black's turn"

### 7.3 Check/Checkmate
1. Move results in check
2. Engine detects check
3. Status updates to "Check!"
4. King square highlights red with pulse
5. Player must move to escape check
6. If no valid moves, status updates to "Checkmate!"
7. Game over, only "New Game" available
